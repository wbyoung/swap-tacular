NEVER COMMIT DIST ON MASTER!
Do: git co -b release master
	(instead of distribution)
	gulp build
	git add dist
	git commit -m "the version" ex. 0.0.4
	git push heroku head:master -f
	(check to see if release worked! if not rollback)
	if worked:
		git tag -a -m 'version number' number (i.e. "0.0.1" 0.0.1)
		git push --tags
		git co master
		git branch -D release 
	if rollback:
		git push heroku tagname:master (i.e. 0.0.1:master)

Dealing with returning multiple promises
// arrays
Promise.resolve() // start promise sequence
.then(function() { return createUser(); })
.then(function(user) { return Promise.all([user, createTokens()]); })
.spead(function(user, tokens) {
})


// scoping
var user;
var tokens;

Promise.resolve() // start promise sequence
.then(function() { return createUser(); })
.then(function(aUser) {
  user = aUser;
  return createTokens();
})
.then(function(theTokens) {
  tokens = theTokens;
})


// bind
Promise.bind({})
.then(function() { return createUser(); })
.then(function(user) {
  this.user = user;
  return createTokens();
})
.then(function(tokens) {
  this.tokens = tokens;
  console.log(this.user);
})
